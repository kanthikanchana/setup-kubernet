Agent side:   Add Linux Host to Monitor:
sudo apt update
sudo apt install nagios-nrpe-server monitoring-plugins
cd /etc/nagios/
vi nrpe_local.cfg
command[check_root]=/usr/lib/nagios/plugins/check_disk -w 20% -c 10% -p /
command[check_ping]=/usr/lib/nagios/plugins/check_ping -H agentip -w 100.0,20% -c 500.0,60% -p 5
command[check_ssh]=/usr/lib/nagios/plugins/check_ssh -4 agentip
command[check_http]=/usr/lib/nagios/plugins/check_http -I agentip
command[check_apt]=/usr/lib/nagios/plugins/check_apt

vi nrpe.cfg
Uncomment the "server_address" line and change the value with the "client01" IP address.
server_address=agentip
One the "allowed_hosts" line, add the Nagios Server IP address "172.16.0.5".
allowed_hosts=127.0.0.1,::1,MasterIP

systemctl restart nagios-nrpe-server
systemctl enable nagios-nrpe-server
systemctl status nagios-nrpe-server

sudo apt install monitoring-plugins nagios-nrpe-plugin
hostname client01
change hostname in /etc/hostname
